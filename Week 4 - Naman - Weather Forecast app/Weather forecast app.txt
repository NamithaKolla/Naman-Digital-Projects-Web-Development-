    <input type="text" id="cityInput" placeholder="Enter city name">

    <div id="weatherDisplay"></div>
  </div>
  <script>
    const apiKey = '19d026384b0467bd7e656735ffba4933'; // <-- get your free key at openweathermap.org and put here

    document.getElementById("getWeather").onclick = function() {
      const city = document.getElementById("cityInput").value.trim();
      if (city === "") {
        document.getElementById("weatherDisplay").innerHTML = "Please enter a city name.";
        return;
      }
      const url = `https://api.openweathermap.org/data/2.5/weather?q=${encodeURIComponent(city)}&appid=${apiKey}&units=metric`;
      fetch(url)
        .then(response => {
          if (!response.ok) throw new Error("City not found.");
          return response.json();
        })
        .then(data => {
          document.getElementById("weatherDisplay").innerHTML = `
            <strong>${data.name}, ${data.sys.country}</strong><br>
            <img src="<https://openweathermap.org/img/wn/${data.weather>[0].icon}@2x.png">
            ${data.weather[0].main} (${data.weather[0].description})<br>
            Temperature: ${data.main.temp}Â°C<br>
            Humidity: ${data.main.humidity}%<br>
            Wind: ${data.wind.speed} m/s
          `;
        })
        .catch(error => {
          document.getElementById("weatherDisplay").innerHTML = error.message;
        });
    };
  </script>
